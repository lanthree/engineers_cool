# 38. 外观数列

[原题链接](https://leetcode-cn.com/problems/count-and-say/)

朴素的模拟思想：

```cpp
class Solution {
public:
    string countAndSay(int n) {
        int now_cnt = 1;
        string now_str = "1";
        while (now_cnt < n) {
            now_cnt++;
            now_str = readstr(now_str);
        }
        return now_str;
    }

private:
    string readstr(string& s) {
        string read_str = "";

        char now_c = s[0];
        int now_cnt = 1;
        for (int i = 1; i < s.length(); ++i) {
            if (s[i] == now_c) {
                now_cnt++;
                continue;
            }
            read_str.append(to_string(now_cnt)).push_back(now_c);
            now_c = s[i];
            now_cnt = 1;
        }
        read_str.append(to_string(now_cnt)).push_back(now_c);
        return read_str;
    }
};
```
