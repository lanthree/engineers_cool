# 242. 有效的字母异位词 

[原题链接](https://leetcode-cn.com/problems/valid-anagram/)

```cpp
class Solution {
public:
    bool isAnagram(string s, string t) {
        if (s.length() != t.length()) return false;

        vector<int> s_cnt(26, 0);
        vector<int> t_cnt(26, 0);
        for (int i = 0; i < s.length(); i++) {
            s_cnt[s[i]-'a']++;
            t_cnt[t[i]-'a']++;
        }

        for (int i = 0; i < 26; i++) {
            if (s_cnt[i] != t_cnt[i])
                return false;
        }
        return true;
    }
};
```
