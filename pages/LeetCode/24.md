# 24. 两两交换链表中的结点

[原题描述](https://leetcode-cn.com/problems/swap-nodes-in-pairs/)

思路：
1. 定义两个指针，指向要交换的两个结点签名
2. 如果可以交换，则操作`first->next / second->next->next`指针进行交换，以及递推
3. 如果不可以交换，则保持现状

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        ListNode* dummy = new ListNode(0, head);
        ListNode* first = head;
        ListNode* second = dummy;

        while(first) {
            second->next = first->next;
            if (second->next) {
                first->next = second->next->next;
                second->next->next = first;
            } else {
                second->next = first;
            }

            second = first;
            first = first->next;
        }

        ListNode* ans = dummy->next;
        delete dummy;
        return ans;
    }
};
```

跟上题一样，我的思路总是递推，官方题解还有一种递归思路（写起来更简单 也更容易理解）：

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        if (head == nullptr || head->next == nullptr) {
            // 至少要有两个结点
            return head;
        }

        ListNode * new_head = head->next;
        head->next = swapPairs(new_head->next);
        new_head->next = head;
        return new_head;
    }
};
```
