# 66. 加一

[原题链接](https://leetcode-cn.com/problems/plus-one/)

从后向前遍历，做加一操作即可：

```cpp
class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {

        int idx = digits.size()-1;
        while (idx >= 0) {
            digits[idx]++;
            if (digits[idx]/10 == 0) // 不能进位直接返回
                return digits;
            // 进位到下一轮
            digits[idx--] %= 10;
        }
   
        reverse(digits.begin(), digits.end());
        digits.push_back(1);
        reverse(digits.begin(), digits.end());
        
        return digits;
    }
};
```
