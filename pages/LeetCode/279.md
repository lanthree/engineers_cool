# 279. 完全平方数 

[原题链接](https://leetcode-cn.com/problems/perfect-squares/)

预处理所有（小于等于10000）的完全平方数都有哪些。然后就是类似零钱问题的DP。

```cpp
class Solution {
    int num_sqs[101];
public:
    Solution() {
        for (int i = 0; i <= 100; i++)
            num_sqs[i] = i*i;
    }

public:
    int numSquares(int n) {
        int dp[10001];
        memset(dp, 0, sizeof(int)*10001);

        for (int i = 0; i <= 100; i++) {
            dp[num_sqs[i]] = 1;
        }

        for (int i = 1; i < n; i++) {
            if (dp[i] == 0)
                continue;

            int sqs_idx = 1;
            while (i+num_sqs[sqs_idx] <= n && sqs_idx <= 100) {
                int sum = i+num_sqs[sqs_idx];
                dp[sum] == 0 ? dp[sum] = dp[i]+1 : dp[sum] = min(dp[sum], dp[i]+1);
                sqs_idx++;
            }
        }

        return dp[n];
    }
};
```

官方题解的写法简化了很多：
1. 没有预先生成所有完全平方数
2. dp值预处理为`INT_MAX`，这样就不用单独处理我这里的`dp[num_sqs[i]] = 1`的初始化，也不用在循环内`min时特判dp[sum] == 0`的情况，厉害了👍

```cpp
class Solution {
public:
    int numSquares(int n) {
        vector<int> f(n + 1);
        for (int i = 1; i <= n; i++) {
            int minn = INT_MAX;
            for (int j = 1; j * j <= i; j++) {
                minn = min(minn, f[i - j * j]);
            }
            f[i] = minn + 1;
        }
        return f[n];
    }
};
```
