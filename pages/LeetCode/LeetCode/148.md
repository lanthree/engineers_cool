# 148. 排序链表

[原题链接](https://leetcode-cn.com/problems/sort-list/)

以`O(log n)`复杂度，而且不能随机访问的话，可以使用归并排序：

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        return _mergeSort(head);
    }

private:
    ListNode* _mergeSort(ListNode* head) {
        if (head == nullptr || head->next == nullptr) {
            return head;
        }

        ListNode* dummy = new ListNode();
        ListNode* slow = head;
        ListNode* quik = head->next;
        while (quik && quik->next) {
            slow = slow->next;
            quik = quik->next->next;
        }
        
        ListNode* last_half = slow->next;
        slow->next = nullptr;
        head = _mergeSort(head); last_half = _mergeSort(last_half);

        dummy->next = nullptr;
        ListNode* tail = dummy;

        while (head && last_half) {
            if (head->val < last_half->val) {
                tail->next = head;
                head = head->next;
            } else {
                tail->next = last_half;
                last_half = last_half->next;
            }
            tail = tail->next;
        }
        while (head) {
            tail->next = head;
            head = head->next;
            tail = tail->next;
        }
        while (last_half) {
            tail->next = last_half;
            last_half = last_half->next;
            tail = tail->next;
        }

        ListNode* merged_head = dummy->next;
        delete dummy;
        return merged_head;
    }
};
```

但是空间复杂度不是`O(1)`，是`O(log n)`，即栈空间。取消栈空间的方法，是自底向上归并，用循环替代递归：

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if (head == nullptr || head->next == nullptr) {
            return head;
        }

        int length = 0;
        ListNode* curr = head;
        while (curr) {
            ++length;
            curr = curr->next;
        }

        for (int sub_length = 1; sub_length < length; sub_length <<= 1) {
            ListNode* dummy = new ListNode();
            ListNode* tail = dummy;
            ListNode* head_1 = nullptr;
            ListNode* head_2 = nullptr;
            ListNode* curr = head;
            while (curr) {
                head_1 = curr;
                curr = _iter(curr, sub_length);
                if (curr == nullptr) {
                    tail->next = head_1;
                    break;
                }

                head_2 = curr->next;
                curr->next = nullptr;

                curr = _iter(head_2, sub_length);
                ListNode* next = nullptr;
                if (curr != nullptr) {
                    next = curr->next;
                    curr->next = nullptr;
                }

                tail->next = _merge(head_1, head_2);
                while(tail->next) {
                    tail = tail->next;
                }
                curr = next;
            }
            head = dummy->next;
            delete dummy;
        }
        return head;
    }

private:
    ListNode* _iter(ListNode* curr, int step) {
        while (curr && --step) {
            curr = curr->next;
        }
        return curr;
    }

    ListNode* _merge(ListNode* a, ListNode* b) {
        ListNode* dummy = new ListNode();
        ListNode* tail = dummy;
        while (a && b) {
            if (a->val < b->val) {
                tail->next = a;
                a = a->next;
            } else {
                tail->next = b;
                b = b->next;
            }
            tail = tail->next;
        }
        while (a) {
            tail->next = a;
            a = a->next;
            tail = tail->next;
        }
        while (b) {
            tail->next = b;
            b = b->next;
            tail = tail->next;
        }
        ListNode* new_head = dummy->next;
        delete dummy;
        return new_head;
    }
};
```
