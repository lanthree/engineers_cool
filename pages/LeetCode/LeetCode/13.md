# 13. 罗马数字转整数 

[原题链接](https://leetcode-cn.com/problems/roman-to-integer/)

```cpp
class Solution {
public:
    int romanToInt(string s) {
        unordered_map<char, int> roman_val = {
            {'I', 1},
            {'V', 5},
            {'X', 10},
            {'L', 50},
            {'C', 100},
            {'D', 500},
            {'M', 1000},
        };
        
        int val = 0;
        int last_roman_val = 1000;
        for (auto c : s) {
            int v = roman_val[c];
            val += v <= last_roman_val ? v : (last_roman_val*-2+v);
            last_roman_val = v;
        }
        return val;
    }
};
```
