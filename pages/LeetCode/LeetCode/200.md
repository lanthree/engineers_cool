# 200. 岛屿数量

[原题链接](https://leetcode-cn.com/problems/number-of-islands/)

用DFS遍历一边即可，每发起一次DFS岛屿数量就加1。

```cpp
class Solution {
    vector<pair<int, int>> dirs_ = {{1,0},{0,1},{-1,0},{0,-1}};
    int m_, n_;
public:
    int numIslands(vector<vector<char>>& grid) {
        m_ = grid.size();
        n_ = grid[0].size();

        int ret = 0;
        vector<vector<bool>> vis(m_, vector<bool>(n_, false));
        for (int i = 0; i < m_; i++) {
            for (int j = 0; j < n_; j++) {
                if (grid[i][j] == '1' && !vis[i][j]) {
                    DFS(grid, i, j, vis);
                    ret += 1;
                }
            }
        }
        return ret;
    }

private:
    void DFS(vector<vector<char>>& grid, int i, int j, vector<vector<bool>>& vis) {
        for (int d = 0; d < 4; d++) {
            int nxt_i = i + dirs_[d].first;
            int nxt_j = j + dirs_[d].second;
            // 是否超出界限
            if (nxt_i < 0 || nxt_i >= m_ || nxt_j < 0 || nxt_j >= n_) {
                continue;
            }
            if (grid[nxt_i][nxt_j] != '1' || vis[nxt_i][nxt_j]) {
                continue;
            }
            vis[nxt_i][nxt_j] = true;
            DFS(grid, nxt_i, nxt_j, vis);
        }
    }
};
```

官方有个并查集的题解，值得学习一下。

```cpp
class UnionFind {
public:
    UnionFind(vector<vector<char>>& grid) {
        count = 0;
        int m = grid.size();
        int n = grid[0].size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == '1') {
                    parent.push_back(i * n + j);
                    ++count;
                }
                else {
                    parent.push_back(-1);
                }
                rank.push_back(0);
            }
        }
    }

    int find(int i) {
        if (parent[i] != i) {
            parent[i] = find(parent[i]);
        }
        return parent[i];
    }

    void unite(int x, int y) {
        int rootx = find(x);
        int rooty = find(y);
        if (rootx != rooty) {
            if (rank[rootx] < rank[rooty]) {
                swap(rootx, rooty);
            }
            parent[rooty] = rootx;
            if (rank[rootx] == rank[rooty]) rank[rootx] += 1;
            --count;
        }
    }

    int getCount() const {
        return count;
    }

private:
    vector<int> parent;
    vector<int> rank;
    int count;
};

class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        int nr = grid.size();
        if (!nr) return 0;
        int nc = grid[0].size();

        UnionFind uf(grid);
        int num_islands = 0;
        for (int r = 0; r < nr; ++r) {
            for (int c = 0; c < nc; ++c) {
                if (grid[r][c] == '1') {
                    grid[r][c] = '0';
                    if (r - 1 >= 0 && grid[r-1][c] == '1') uf.unite(r * nc + c, (r-1) * nc + c);
                    if (r + 1 < nr && grid[r+1][c] == '1') uf.unite(r * nc + c, (r+1) * nc + c);
                    if (c - 1 >= 0 && grid[r][c-1] == '1') uf.unite(r * nc + c, r * nc + c - 1);
                    if (c + 1 < nc && grid[r][c+1] == '1') uf.unite(r * nc + c, r * nc + c + 1);
                }
            }
        }

        return uf.getCount();
    }
};
```
