# 12. 整数转罗马数字 

[原题链接](https://leetcode-cn.com/problems/integer-to-roman/)

```cpp
class Solution {
public:
    string intToRoman(int num) {
        string roman;
        vector<pair<int, string>> case_reps = {
            {1000, "M"},
            {900, "CM"},
            {500, "D"},
            {400, "CD"},
            {100, "C"},
            {90, "XC"},
            {50, "L"},
            {40, "XL"},
            {10, "X"},
            {9, "IX"},
            {5, "V"},
            {4, "IV"},
            {1, "I"}
        };
        for (auto case_rep : case_reps) {
            while(num >= case_rep.first) {
                num -= case_rep.first;
                roman += case_rep.second;
            }
        }
        return roman;
    }
};
```
