# 134. 加油站

[原题链接](https://leetcode-cn.com/problems/gas-station/)

```cpp
class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int total = 0, sub_total = 0, res = 0;
        int len = gas.size();
        for (int i = 0; i < len; ++i) {
            total += gas[i] - cost[i];
            sub_total += gas[i] - cost[i];
            if (sub_total < 0) {
                res = i+1;
                sub_total = 0;
            }
        }

        if (total < 0)
            return -1;

        return res;
    }
};
```
