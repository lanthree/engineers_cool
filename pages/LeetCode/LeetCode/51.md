# 51. N 皇后

[原题链接](https://leetcode-cn.com/problems/n-queens/)

DFS搜索即可

```cpp
class Solution {
    vector<vector<string>> ans;
public:
    vector<vector<string>> solveNQueens(int n) {

        vector<string> q(n, string(n, '.'));
        DFS(n, 0, q);
        // ans.push_back(q);
        return ans;
    }

private:
    void DFS(int n, int d, vector<string>& q) {
        if (n == d) {
            ans.push_back(q);
            return;
        }

        int i = d;
        for (int j = 0; j < n; j++) {
            if (isValid(n, q, i, j)) {
                q[i][j] = 'Q';
                DFS(n, d+1, q);
                q[i][j] = '.';
            }
        }
    }

    bool isValid(int n, vector<string>& q, int i, int j) {
        for (int l = 1; l <= i; ++l) {
            if (q[i-l][j] == 'Q') {
                return false;
            }
            int n_j = j-l;
            if (n_j >= 0 && n_j < n && q[i-l][n_j] == 'Q') {
                return false;
            }
            n_j = j+l;
            if (n_j >= 0 && n_j < n && q[i-l][n_j] == 'Q') {
                return false;
            }
        }
        return true;
    }
};
```
