# 155. 最小栈

[原题链接](https://leetcode-cn.com/problems/min-stack/)

维护一个辅助栈：

```cpp
clclass MinStack {
public:
    /** initialize your data structure here. */
    MinStack() {
        min_st.push(numeric_limits<int>::max());
    }
    
    void push(int val) {
        st.push(val);
        min_st.push(min(val, min_st.top()));
    }
    
    void pop() {
        st.pop();
        min_st.pop();
    }
    
    int top() {
        return st.top();
    }
    
    int getMin() {
        return min_st.top();
    }

private:

private:
    stack<int> st;
    stack<int> min_st;
};
```
