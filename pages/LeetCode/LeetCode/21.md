# 21. 合并两个有序链表

[原题描述](https://leetcode-cn.com/problems/merge-two-sorted-lists/)

基于归并排序的思想，merge两个链表即可：

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* dummy = new ListNode(0, nullptr);
        ListNode* merge_tail = dummy;

        while (l1 || l2) {
            int v1 = l1 ? l1->val : INT_MAX;
            int v2 = l2 ? l2->val : INT_MAX;
            if (v1 < v2) {
                merge_tail->next = l1;
                merge_tail = merge_tail->next;
                l1 = l1->next;
            } else {
                merge_tail->next = l2;
                merge_tail = merge_tail->next;
                l2 = l2->next;
            }
        }

        ListNode* ans = dummy->next;
        delete dummy;
        return ans;
    }
};
```

但是有优化点：
1. merge的while循环 可以是条件&&，循环推出后，直接把没空的链表接到merge_tail即可：

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* dummy = new ListNode(0, nullptr);
        ListNode* merge_tail = dummy;

        while (l1 && l2) {
            if (l1->val < l2->val) {
                merge_tail->next = l1;
                merge_tail = merge_tail->next;
                l1 = l1->next;
            } else {
                merge_tail->next = l2;
                merge_tail = merge_tail->next;
                l2 = l2->next;
            }
        }
        if (l1) {
            merge_tail->next = l1;
        } else if (l2) {
            merge_tail->next = l2;
        }

        ListNode* ans = dummy->next;
        delete dummy;
        return ans;
    }
};
```

另外，这个题还可以使用递归的思路：

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        if (l1 == nullptr) {
            return l2;
        } else if (l2 == nullptr) {
            return l1;
        }

        if (l1->val < l2->val) {
            l1->next = mergeTwoLists(l1->next, l2);
            return l1;
        } else {
            l2->next = mergeTwoLists(l1, l2->next);
            return l2;
        }
    }
};
```
