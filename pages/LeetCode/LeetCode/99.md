# 99. 恢复二叉搜索树 

[原题链接](https://leetcode-cn.com/problems/recover-binary-search-tree/)

先中序遍历一边，拿到中序遍历的顺序，然后遍历数组找到需要交换的位置；最后再遍历树并对相应的结点赋值。

如果不想遍历两边树的话，感觉可以在第一次遍历的时候，存下结点地址，而不是结点值。

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void recoverTree(TreeNode* root) {
        vector<int> nums;
        inorder(root, nums);
        pair<int, int> swapped = findTwoSwapped(nums);
        int re_cnt = 2;
        recover(root, re_cnt, swapped.first, swapped.second);
    }

private:
    void inorder(TreeNode* node, vector<int>& nums) {
        if (!node) return;
        inorder(node->left, nums);
        nums.push_back(node->val);
        inorder(node->right, nums);
    }

    pair<int, int> findTwoSwapped(vector<int>& nums) {
        int x = -1, y = -1;
        for (int i = 0; i < nums.size()-1; i++) {
            if (nums[i] > nums[i+1]) {
                x = nums[i+1];
                if (y == -1) {
                    y = nums[i];
                } else {
                    break;
                }
            }
        }
        return {x, y};
    }

    void recover(TreeNode* node, int& re_cnt, int x, int y) {
        if (!node) return;
        if (re_cnt == 0) return;

        if (node->val == x) {
            node->val = y;
            re_cnt--;
        } else if (node->val == y) {
            node->val = x;
            re_cnt--;
        }

        recover(node->left, re_cnt, x, y);
        recover(node->right, re_cnt, x, y);
    }
};
```

因为在遍历数组找需要交换的结点时，只用到了前一个结点，那么可以不增加这个数组，只维护前一结点即可。

另外，下面用stack模拟递归栈：

```cpp
class Solution {
public:
    void recoverTree(TreeNode* root) {
        stack<TreeNode*> stk;
        TreeNode* x = nullptr;
        TreeNode* y = nullptr;
        TreeNode* pre_node = nullptr;

        while (!stk.empty() || root) {
            while (root) {
                stk.push(root);
                root = root->left;
            }
            root = stk.top();
            stk.pop();

            if (pre_node && pre_node->val > root->val) {
                y = root;
                if (x == nullptr)
                    x = pre_node;
                else
                    break;
            }

            pre_node = root;
            root = root->right;
        }
        
        swap(x->val, y->val);
    }
};
```
