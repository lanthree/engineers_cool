# 140. 单词拆分 II

[原题链接](https://leetcode-cn.com/problems/word-break-ii/)

字典树 + DFS即可：

```cpp
class Solution {
    struct TriNode {
        TriNode() : children(26, nullptr), is_word(false) {}
        vector<TriNode*> children;
        bool is_word;
    };

    TriNode root;
    vector<string> solutions;
public:
    vector<string> wordBreak(string s, vector<string>& wordDict) {
        _buildByDict(wordDict);
        // _printDict();

        string solution;
        _dfs(solution, s, 0);
        return solutions;
    }

private:
    void _dfs(string solution, string& s, int idx) {
        if (idx == s.length()) {
            solutions.push_back(solution);
            return;
        }

        TriNode* node = &root;
        for (int i = idx; i < s.length(); ++i) {
            int c_idx = s[i]-'a';
            if (node->children[c_idx] == nullptr) {
                return;
            }
            solution.push_back(s[i]);
            node = node->children[c_idx];
            if (node->is_word) {
                if (i != s.length()-1)
                    solution.push_back(' ');
                _dfs(solution, s, i+1);
                if (i != s.length()-1)
                    solution.pop_back();
            }
        }
    }


    void _buildByDict(vector<string>& wordDict) {
        for (auto word : wordDict) {
            _build(root, word, 0);
        }
    }

    void _printDict() {
        string word;
        _print_dict(root, word);
    }

    void _build(TriNode& node, string& word, int idx) {
        if (idx == word.size()) {
            node.is_word = true;
            return;
        }

        char c_idx = word[idx] - 'a';
        if (node.children[c_idx] == nullptr) {
            node.children[c_idx] = new TriNode();
        }
        _build(*node.children[c_idx], word, idx+1);
    }

    void _print_dict(TriNode& node, string& word) {
        if (node.is_word) {
            cout << word << endl;
        }
        for (int i = 0; i < 26; ++i) {
            if (node.children[i] == nullptr) {
                continue;
            }

            word.push_back('a'+i);
            _print_dict(*node.children[i], word);
            word.pop_back();
        }
    }
};
```

另外可以用记忆数组减少重复计算：

```cpp
class Solution {
    struct TriNode {
        TriNode() : children(26, nullptr), is_word(false) {}
        vector<TriNode*> children;
        bool is_word;
    };

    TriNode root;
    unordered_map<int, vector<string>> solutions;
public:
    vector<string> wordBreak(string s, vector<string>& wordDict) {
        _buildByDict(wordDict);

        _dfs(s, 0);
        return solutions[0];
    }

private:
    void _dfs(string& s, int idx) {
        if (solutions.count(idx)) {
            return;
        }

        if (idx == s.length()) {
            solutions[idx] = {""};
            return;
        }

        TriNode* node = &root;
        string word;
        for (int i = idx; i < s.length(); ++i) {
            int c_idx = s[i]-'a';
            if (node->children[c_idx] == nullptr) {
                return;
            }
            word.push_back(s[i]);
            node = node->children[c_idx];
            if (node->is_word) {
                _dfs(s, i+1);
                for (auto& sub_sol : solutions[i+1]) {
                    solutions[idx].push_back( word + (sub_sol.empty() ? "" : " " + sub_sol));
                }
            }
        }
    }


    void _buildByDict(vector<string>& wordDict) {
        for (auto word : wordDict) {
            _build(root, word, 0);
        }
    }

    void _build(TriNode& node, string& word, int idx) {
        if (idx == word.size()) {
            node.is_word = true;
            return;
        }

        char c_idx = word[idx] - 'a';
        if (node.children[c_idx] == nullptr) {
            node.children[c_idx] = new TriNode();
        }
        _build(*node.children[c_idx], word, idx+1);
    }
};
```

最后，字典树在这里有点大才小用了，可以直接用集合查找：

```cpp
class Solution {
    unordered_map<int, vector<string>> solutions;
    unordered_set<string> words;
public:
    vector<string> wordBreak(string s, vector<string>& wordDict) {
        words.insert(wordDict.begin(), wordDict.end());
        _dfs(s, 0);
        return solutions[0];
    }

private:
    void _dfs(string& s, int idx) {
        if (solutions.count(idx)) {
            return;
        }

        if (idx == s.length()) {
            solutions[idx] = {""};
            return;
        }

        for (int i = idx+1; i <= s.length(); ++i) {
            string word = s.substr(idx, i-idx);
            if (words.count(word)) {
                _dfs(s, i);
                for (auto& sub_sol : solutions[i]) {
                    solutions[idx].push_back(word + (sub_sol.empty() ? "" : " " + sub_sol));
                }
            }
        }
    }
};
```
