# 202. 快乐数

[原题链接](https://leetcode-cn.com/problems/happy-number/)

直接模拟接口，另外拿一个容器记录已遍历的数字：

```cpp
class Solution {
public:
    bool isHappy(int n) {
        set<int> vis;
        while (true) {
            int nxt_n = powN(n);
            cout << nxt_n << endl;
            if (nxt_n == 1) {
                return true;
            } else if (vis.count(nxt_n)) {
                return false;
            } else {
                vis.insert(nxt_n);
                n = nxt_n;
            }
        }
        return false;
    }

private:
    int powN(int n) {
        int nxt = 0;
        while (n) {
            int sub_n = n%10;
            n /= 10;
            nxt += sub_n*sub_n;
        }
        return nxt;
    }
};
```
